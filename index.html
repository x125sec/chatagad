Firestore Security Rules

These rules have been applied to fix permission errors:

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow users to manage their own queue entries
    match /queue/{userId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.auth.uid == userId;
    }

    // Allow chat creation and updates
    match /chats/{chatId} {
      allow create: if request.auth != null;
      allow update, read: if 
        request.auth != null && 
        (resource.data.user1 == request.auth.uid || 
         resource.data.user2 == request.auth.uid);
    }
  }
}
